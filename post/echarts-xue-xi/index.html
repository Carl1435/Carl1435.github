echarts学习

2020-11-16 1 min read 
var myChart = echarts.init(document.getElementById('main'));
    
    // 指定图表的配置项和数据
    var option = {
    //图表标题
        title: {
            text: '第一个 ECharts 实例'
        },
        提示信息
        tooltip: {},
        图例组件
        legend: {
            data:['销量']
        },
        x轴
        xAxis: {
            data: ["衬衫","羊毛衫","雪纺衫","裤子","高跟鞋","袜子"]
        },
        y轴
        yAxis: {},
        每个系列
        series: [{
            name: '销量',
            type: 'bar',
            data: [5, 20, 36, 10, 10, 20]
        }]
    };

    // 使用刚指定的配置项和数据显示图表。
    myChart.setOption(option);
